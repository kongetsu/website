import javax.swing.JFrame;
import java.awt.Canvas;
import java.awt.Color;
import java.awt.Graphics;
import java.awt.Dimension;

public class AccurateStarDrawing extends Canvas
{
   
   public void paint(Graphics g)
   {
      g.setColor(Color.YELLOW);
      
      int centerX = 100;
      int centerY = 100;
      int outerRadius = 50;
      int innerRadius = outerRadius / 2;
      
      double angleIncrement = 2 * Math.PI / 5;
      
      int[] xPoints = new int[10];
      int[] yPoints = new int[10];
      
      for (int i = 0; i < 5; i++)
      {
         double angle = i * 2 * angleIncrement;
         xPoints[2 * i] = (int) (centerX + outerRadius * Math.cos(angle));
         yPoints[2 * i] = (int) (centerY + outerRadius * Math.sin(angle));
         
         angle += angleIncrement;
         xPoints[2 * i + 1] = (int) (centerX + innerRadius * Math.cos(angle));
         yPoints[2 * i + 1] = (int) (centerY + innerRadius * Math.sin(angle));
      }
      
      g.fillPolygon(xPoints, yPoints, 10);
   }
   
   public static void main(String[] args)
   {
      JFrame frame = new JFrame("Accurate Yellow Five-Pointer Star");
      AccurateStarDrawing canvas = new AccurateStarDrawing();
      canvas.setPreferredSize(new Dimension(200, 200));
      
      frame.add(canvas);
      frame.pack();
      frame.setVisible(true);
      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
   }
}

public class FlagStarCoordinates
{
   
   /**
       * 计算五星红旗上星星顶点坐标的方法
       *
       * @param largeone 是否为大星，true表示大星，false表示小星
       * @param starOrder 小星的顺序（1 - 4），如果是大星则此参数无实际意义
       * @return 一个10 x 2的二维数组，代表10个顶点（5个外顶点和5个内顶点）的坐标，第一维表示顶点序号，第二维0表示x坐标，1表示y坐标
       */
   public static int[][] starCordinate(boolean largeone, int starOrder)
   {
      int width = 1152; // 设定旗宽像素
      int height = width * 2 / 3; // 根据长宽比3:2计算旗高像素
      
      // 大星外接圆直径（基于旗高）
      int largeDiameter = height * 3 / 10;
      // 小星外接圆直径（基于旗高）
      int smallDiameter = height * 1 / 10;
      
      // 大星圆心坐标（基于旗面的相对位置计算）
      int largeCenterX = width * 5 / 10;
      int largeCenterY = height * 5 / 10;
      
      // 小星圆心坐标数组（4颗小星的圆心坐标）
      int[][] smallCenters =
      {
         {
            width * 10 / 10, height * 2 / 10
         }
         ,
         {
            width * 12 / 10, height * 4 / 10
         }
         ,
         {
            width * 12 / 10, height * 7 / 10
         }
         ,
         {
            width * 10 / 10, height * 9 / 10
         }
      };
      
      int[][] coordinates = new int[10][2];
      if (largeone)
      {
         // 计算大星的顶点坐标
         int radius = largeDiameter / 2;
         double angleStep = 2 * Math.PI / 5;
         for (int i = 0; i < 5; i++)
         {
            double angle = i * angleStep;
            // 外顶点x坐标
            coordinates[i][0] = (int) (largeCenterX + radius * Math.cos(angle));
            // 外顶点y坐标
            coordinates[i][1] = (int) (largeCenterY - radius * Math.sin(angle));
            
            double innerAngle = angle + angleStep / 2;
            // 内顶点x坐标
            coordinates[i + 5][0] = (int) (largeCenterX + radius / 2 * Math.cos(innerAngle));
            // 内顶点y坐标
            coordinates[i + 5][1] = (int) (largeCenterY - radius / 2 * Math.sin(innerAngle));
         }
      }
      else
      {
         if (starOrder >= 1 && starOrder <= 4)
         {
            // 计算对应序号小星的顶点坐标
            int radius = smallDiameter / 2;
            int centerX = smallCenters[starOrder - 1][0];
            int centerY = smallCenters[starOrder - 1][1];
            double angleStep = 2 * Math.PI / 5;
            for (int i = 0; i < 5; i++)
            {
               double angle = i * angleStep;
               // 外顶点x坐标
               coordinates[i][0] = (int) (centerX + radius * Math.cos(angle));
               // 外顶点y坐标
               coordinates[i][1] = (int) (centerY - radius * Math.sin(angle));
               
               double innerAngle = angle + angleStep / 2;
               // 内顶点x坐标
               coordinates[i + 5][0] = (int) (centerX + radius / 2 * Math.cos(innerAngle));
               // 内顶点y坐标
               coordinates[i + 5][1] = (int) (centerY - radius / 2 * Math.sin(innerAngle));
            }
         }
      }
      return coordinates;
   }

public class FiveStarFlag extends JPanel

   public static Shape createPentacle(double sx, double sy, double radius, double theta)
   {
      final double arc = Math.PI / 5;
      final double rad = Math.sin(Math.PI / 10) / Math.sin(3 * Math.PI / 10);
      GeneralPath path = new GeneralPath();
      path.moveTo(1, 0);
      for (int i = 0; i < 5; i++)
      {
         path.lineTo(rad * Math.cos((1 + 2 * i) * arc), rad * Math.sin((1 + 2 * i) * arc));
         path.lineTo(Math.cos(2 * (i + 1) * arc), Math.sin(2 * (i + 1) * arc));
      }
      path.closePath();
      AffineTransform atf = AffineTransform.getScaleInstance(radius, radius);
      atf.translate(sx / radius, sy / radius);
      atf.rotate(theta);
      return atf.createTransformedShape(path);
   }
   
   private int width, height;
   private double maxR = 0.15, minR = 0.05;
   private double maxX = 0.25, maxY = 0.25;
   private double[] minX =
   {
      0.50, 0.60, 0.60, 0.50
   };
   private double[] minY =
   {
      0.10, 0.20, 0.35, 0.45
   };
   
   public FiveStarFlag(int width)
   {
      this.width = width / 3 * 3;
      this.height = width / 3 * 2;
      setPreferredSize(new Dimension(this.width, this.height));
   }
   
   @Override
   protected void paintComponent(Graphics g)
   {
      Graphics2D graphics2D = (Graphics2D) g;
      
      graphics2D.setPaint(Color.RED);
      graphics2D.fillRect(0, 0, width, height);
      
      double ox = height * maxX, oy = height * maxY;
      graphics2D.setPaint(Color.YELLOW);
      graphics2D.fill(createPentacle(ox, oy, height * maxR, -Math.PI / 2));
      
      for (int i = 0; i < 4; i++)
      {
         double sx = minX[i] * height, sy = minY[i] * height;
         double theta = Math.atan2(oy - sy, ox - sx);
         graphics2D.fill(createPentacle(sx, sy, height * minR, theta));
      }
   }
}

import java.util.Scanner;

public class SumNumbers
{
   public static void main(String[] args)
   {
      Scanner scanner = new Scanner(System.in);
      
      // Prompt the user to enter numbers
      System.out.println("Enter two or more numbers separated by spaces:");
      
      // Read the entire line of input
      String inputLine = scanner.nextLine();
      
      // Split the input by spaces into an array of strings
      String[] inputs = inputLine.split(" ");
      
      double sum = 0; // Variable to store the sum of numbers
      
      // Iterate over each input and try to parse it as a number
      for (String input : inputs)
      {
         try
         {
            // Try to convert the input string to a number (double)
            double number = Double.parseDouble(input);
            sum += number; // Add the valid number to the sum
            
         }
         catch (NumberFormatException e)
         {
            // If parsing fails, print "NaN" and exit the program
            System.out.println("NaN");
            scanner.close(); // Close the scanner before exiting
            return;
         }
      }
      
      // If all inputs were valid numbers, print the sum
      System.out.println("The sum is: "+ sum);
      
      scanner.close(); // Close the scanner at the end
      
   }
}
